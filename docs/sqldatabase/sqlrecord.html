164        alias: str,

166        """Parse an alias into its components.
167
168        Args:
169            alias (str): The alias to parse.
170
171        Returns:
172            tuple[str | None, str | None, str | None]: The function name, table fully qualified name, and column name.
173        """
174        function_name = table_fully_qualified_name = column_name = None

178            function_name = alias_parts[function_identifier_index + 1]

181            column_fully_qualified_name_parts = alias_parts[
182                column_identifier_index + 1 :
183            ]
184            table_fully_qualified_name = ".".join(
185                column_fully_qualified_name_parts[:-1]
186            )
187            column_name = column_fully_qualified_name_parts[-1]

192        cls, alias: str, database: SqlDatabase
193    ) -> SqlColumn | SqlAggregateFunction:
194        """Get an item by its alias.
195
196        Args:
197            alias (str): The alias of the item.
198            database (SqlDatabase): The database instance.
199
200        Returns:
201            SqlColumn | SqlAggregateFunction: The item associated with the alias.
202        """
203        function_name, table_fully_qualified_name, column_name = cls._parse_alias(alias)
204        if table_fully_qualified_name is not None and column_name is not None:

206                column_name

208        else:
209            column = None
210        if function_name is not None:
211            function_class = database.functions(function_name)
212            assert not (
213                column is None
214                and issubclass(function_class, SqlAggregateFunctionWithMandatoryColumn)
215            ), f"Unexpected alias format: {alias}"
216            item: SqlColumn | SqlAggregateFunction = function_class(column)
217        elif column is not None:
218            item = column
219        else:
220            assert False, f"Unexpected alias format: {alias}"
221        return item
